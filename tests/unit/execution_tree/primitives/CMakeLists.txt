    ${${test}_FLAGS}
    EXCLUDE_FROM_ALL
    FOLDER "Tests/Unit/ExecutionTree/Primitives")

  add_phylanx_pseudo_target(tests.unit.execution_tree.primitives.${test})

    tests.unit.execution_tree.primitives.${test})
